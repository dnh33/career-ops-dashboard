```python
_REPORT_LOCK = Path("/tmp/career-ops-report.lock")

def _next_report_path(company_slug: str) -> Path:
    reports_dir = career_ops_root() / "reports"
    reports_dir.mkdir(parents=True, exist_ok=True)

        for f in reports_dir.glob("[0-9][0-9][0-9]-*.md"):
            m = re.match(r"^(\d{3})-", f.name)
```

**Objective:** Create the Cursor CLI wrapper using the correct `--print` flag for non-interactive use.

```python
"""Cursor CLI agent wrapper — runs tasks via cursor --print."""
from __future__ import annotations
import asyncio
import shutil
from pathlib import Path


class CursorError(Exception):
    pass


class CursorAgent:
    def __init__(self, workdir: str | Path = Path("/opt/career-ops"), timeout: int = 300):
        self.workdir = Path(workdir)
        self.timeout = timeout

    async def run(self, prompt: str) -> str:
        """Run a task via Cursor CLI in non-interactive mode."""
        if not shutil.which("cursor"):
            raise CursorError("cursor CLI not found — run: npm install -g cursor")

        try:
            proc = await asyncio.create_subprocess_exec(
                "cursor", "--print", "--model", "auto", prompt,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
                cwd=str(self.workdir),
            )

            stdout, stderr = await asyncio.wait_for(
                proc.communicate(),
                timeout=self.timeout,
            )

            if proc.returncode != 0:
                raise CursorError(f"cursor exited {proc.returncode}: {stderr.decode()[:500]}")

            output = stdout.decode().strip()
            if not output:
                raise CursorError("cursor returned empty output")
            return output

        except asyncio.TimeoutError:
            proc.kill()
            await proc.wait()
            raise CursorError(f"cursor timed out after {self.timeout}s")
```

```python
import os, httpx

OPENROUTER_KEY = os.environ.get("OPENROUTER_API_KEY", "")
MODEL = "owl-alpha"


async def _call(system: str, user: str, max_tokens=1500) -> str:
    async with httpx.AsyncClient(timeout=20) as c:
        r = await c.post(
            "https://openrouter.ai/api/v1/chat/completions",

            json={"model": MODEL, "messages": [

async def suggest_role_title(jd_text: str) -> str:
```

```python
async def submit_evaluation(req: EvaluateRequest):
    if not req.text and not req.url:
        raise HTTPException(status_code=400, detail="Provide either 'text' or 'url'")
    if not req.text:
        raise HTTPException(status_code=202, detail="URL input not fetched server-side. Retrieve page text and re-POST with {'text': ...}")

            report=f"⚠️ **Potential scam detected:** {legitimacy}\n\nEvaluation skipped.",
            company=company_raw, role="", report_path=""
        )
    
    # 3. Full evaluation via Cursor (expensive)
    agent = CursorAgent(timeout=300)
    mode_context = load_system_prompt()

## Job Description
{jd_text[:8000]}

## Company (extracted): {company_raw}
## Legitimacy: {legitimacy}

Perform full A-G evaluation. Return the complete report as your response."""

    return EvaluateResponse(
        report=report, company=company_raw, role="", report_path=str(report_path)
    )
```

```python
"""Output router — PDF generation and listing with path traversal protection."""
from pathlib import Path
from fastapi import APIRouter, HTTPException
from fastapi.responses import FileResponse
from app.config import OUTPUT_DIR

router = APIRouter(tags=["output"])


def _safe_path(filename: str, base_dir: Path) -> Path:
    """Resolve filename within base_dir, reject traversal attempts."""
    # Reject any path separators or parent references
    if "/" in filename or "\\" in filename or ".." in filename:
        raise HTTPException(status_code=400, detail="Invalid filename")
    
    target = (base_dir / filename).resolve()
    if not target.is_relative_to(base_dir.resolve()):
        raise HTTPException(status_code=403, detail="Path traversal detected")
    
    if not target.exists():
        raise HTTPException(status_code=404, detail="File not found")
    
    return target


@router.get("/output")
async def list_output():
    """List all generated PDF files."""
    files = []
    if OUTPUT_DIR.exists():
        for f in sorted(OUTPUT_DIR.iterdir(), key=lambda x: x.stat().st_mtime, reverse=True):
            if f.suffix.lower() == ".pdf":
                files.append({
                    "id": f.stem,
                    "filename": f.name,
                    "size_bytes": f.stat().st_size,
                    "created_at": datetime.fromtimestamp(f.stat().st_mtime).isoformat(),
                })
    return {"files": files}


@router.post("/output/generate")
async def generate_pdf(body: dict):
    """Generate PDF from report via Cursor."""
    report_id = body.get("report_id", "")
    if not report_id or not report_id.isdigit():
        raise HTTPException(status_code=400, detail="Invalid report_id")
    
    agent = CursorAgent(timeout=120)
    result = await agent.run(
        f"Generate PDF for report {report_id}. Read reports/ and cv.md, "
        f"run generate-pdf.mjs, save to output/. Return the filename."
    )
    return {"status": "complete", "result": result}


@router.get("/output/download/{filename}")
async def download_pdf(filename: str):
    """Download a PDF — with path traversal protection."""
    target = _safe_path(filename, OUTPUT_DIR)
    return FileResponse(target, media_type="application/pdf",
                        headers={"Content-Disposition": f"attachment; filename={filename}"})


@router.get("/output/view/{filename}")
async def view_pdf(filename: str):
    """View a PDF inline — with path traversal protection."""
    target = _safe_path(filename, OUTPUT_DIR)
    return FileResponse(target, media_type="application/pdf")
```

```python
async def get_report_by_tracker(tracker_number: int):
    """Find report by tracker number (stored in frontmatter)."""
    for f in REPORTS_DIR.iterdir():
        if f.suffix == ".md":
            content = f.read_text()
            if f"tracker_id: {tracker_number}" in content:
                m = FILENAME_RE.match(f.name)
                if m:
                    return get_single_report(m.group(1))
    raise HTTPException(404, f"No report for tracker #{tracker_number}")
```
